TunSafe

Ubuntu 18.04

client

1.Download 

$ git clone https://github.com/TunSafe/TunSafe.git
$ cd TunSafe

2.Install compiler environment

$ apt install clang-6.0 
$ apt install make

3.install TunSafe

$ make && make install

4.Run TunSafe

# 1.start
$ tunsafe start -d TunSafe.conf

# 2.stop
$ tunsafe stop tun0

CentOS 需测试

refernece

llvm-toolset-6.0

devtoolset-8-gcc

gcc-rpm

1.配置repo

/etc/yum.repos.d/springdale-7-SCL.repo

[SCL-core]
name=Springdale SCL Base $releasever - $basearch
mirrorlist=http://springdale.princeton.edu/data/springdale/SCL/$releasever/$basearch/mirrorlist
#baseurl=http://springdale.princeton.edu/data/springdale/SCL/$releasever/$basearch
gpgcheck=1
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-springdale

2.install

yum install centos-release-scl
yum install devtoolset-8 llvm-toolset-6.0
yum install llvm-toolset-6.0-clang-analyzer llvm-toolset-6.0-clang-tools-extra
scl enable devtoolset-8 llvm-toolset-6.0 bash

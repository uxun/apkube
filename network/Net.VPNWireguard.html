官方：https://www.wireguard.com/install/

Kernel：linux >= 3.10

1.0 Centos install

#1.upgrade kernel 需要更新kernel

curl -Lo /etc/yum.repos.d/wireguard.repo https://copr.fedorainfracloud.org/coprs/jdoss/wireguard/repo/epel-7/jdoss-wireguard-epel-7.repo

1.1 Configuration

# Server config 
$ umask 077
# 生成私钥，公钥，共享密钥
$ wg genkey | tee privatekey | wg pubkey > publickey | wg genpsk > preshared
# conf
$ mkdir /etc/wireguard/

1.2 Server

$ cat > /etc/wireguard/wg0.conf << EOF
#server
[Interface]
Address = 10.0.0.1/24
SaveConfig = true
PrivateKey = <insert server_private_key>
ListenPort = 51820
PostUp = iptables -A FORWARD -i wg0 -j ACCEPT; iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE
PostDown = iptables -D FORWARD -i wg0 -j ACCEPT; iptables -t nat -D POSTROUTING -o eth0 -j MASQUERADE

[Peer]
#client test
PublicKey = <insert client_public_key>
PresharedKey = <insert preshared>
AllowedIPs = 10.0.0.2/32
EOF

# start service 
$ systemctl start wg-quick@wg0

1.3 Client

# Client config  
$ mkdir /etc/wireguard/
$ cat > /etc/wireguard/wg0.conf << EOF
[Interface]
Address = 10.0.0.2/24
PrivateKey = [client's PRIVATE KEY]
DNS = 10.0.0.1

[Peer]
PublicKey = [SERVER PUBLICKEY]
PresharedKey = [PRE-SHARED KEY]
AllowedIPs = 0.0.0.0/0
Endpoint = <Server Public IP>:51820 
EOF

# start service 
$ systemctl start wg-quick@wg0

2.0 Demo Server

配置脚本 client.sh

Info

#!/bin/bash
# SPDX-License-Identifier: GPL-2.0
#
# Copyright (C) 2015-2018 Jason A. Donenfeld <Jason@zx2c4.com>. All Rights Reserved.

set -e
[[ $UID == 0 ]] || { echo "You must be root to run this."; exit 1; }
exec 3<>/dev/tcp/demo.wireguard.com/42912
privatekey="$(wg genkey)"
wg pubkey <<<"$privatekey" >&3
IFS=: read -r status server_pubkey server_port internal_ip <&3
[[ $status == OK ]]
ip link del dev wg0 2>/dev/null || true
ip link add dev wg0 type wireguard
wg set wg0 private-key <(echo "$privatekey") peer "$server_pubkey" allowed-ips 0.0.0.0/0 endpoint "demo.wireguard.com:$server_port" persistent-keepalive 25
ip address add "$internal_ip"/24 dev wg0
ip link set up dev wg0
if [ "$1" == "default-route" ]; then
    host="$(wg show wg0 endpoints | sed -n 's/.*\t\(.*\):.*/\1/p')"
    ip route add $(ip route get $host | sed '/ via [0-9]\{1,3\}\.[0-9]\{1,3\}\.[0-9]\{1,3\}\.[0-9]\{1,3\}/{s/^\(.* via [0-9]\{1,3\}\.[0-9]\{1,3\}\.[0-9]\{1,3\}\.[0-9]\{1,3\}\).*/\1/}' | head -n 1) 2>/dev/null || true
    ip route add 0/1 dev wg0
    ip route add 128/1 dev wg0
fi

2.1 Command line configuration (对等配置)

#start server
$ umask 077
$ wg genkey | tee privatekey | wg pubkey > publickey

$ ip link add wg0 type wireguard
$ ip addr add 10.0.0.1/24 dev wg0
$ wg set wg0 private-key /root/wireguard/privatekey
$ ip link set wg0 up
$ wg set wg0 peer CLIENT_PUBKEY allowed-ips 10.0.0.2/32 endpoint 192.168.1.179:55710

#start client
yum install wireguard-dkms wireguard-tools -y
$ umask 077

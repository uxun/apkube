Redis CMD

常用管理命令

redis-cli -h
-h:指定服务器地址
-p:指定端口
-s:指定socket位置
-a:指定密码

redis-cli -p 6380 

echo "keys *" redis-cli -p 6380 -x | grep 'ip'

echo "del lgg:rate:day:count:20190713:219.134.216.111" | redis-cli -p 6380 -x

echo "ttl lgg:rate:day:count:20190713:219.134.216.111" | redis-cli -p 6380 -x

echo "get lgg:rate:day:count:20190713:219.134.216.111" | redis-cli -p 6380 -x

echo "set white_IP 1" | redis-cli -p 6380 -x

echo "expire lgg:rate:day:count:20190713:219.134.216.111 3600" | redis-cli -p 6380 -x

interactive

# 关于服务的信息
>help @server
 CLIENT LIST    # client connect info
 CLIENT KILL    # kill client process
 SLAVEOF        # masterIP:port 可作为master从 

# 获取数组中单个指令帮助信息
>help append

# 查询数据类型
>help @string | @hash | @list

component

redis-benchmark(性能压测)
    redis-benchmark -q -n 100000
    redis-benchmark -q -n 100000 -p 6380
redis-check-aof(检查Redis持久化之后的文件中(AOF/RDB)是否出现错误)

other

delete all keys in redis

redis-cli -p 6380 flushall

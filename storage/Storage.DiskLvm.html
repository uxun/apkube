LVM

LVM 扩容

硬盘>物理卷>卷组>逻辑卷
Disk->PV->VG->LV

1.创建Disk(LVM)

$ fdisk /dev/sdb --- > n > p > t (8e lvm) > w
$ mkfs.xfs /dev/sdb1

E.g

[root@alpha-ops lib]# fdisk /dev/sda

   e   extended

2.让内核重读分区识别LVM

[root@alpha-ops lib]# partx -a /dev/sda
partx: /dev/sda: error adding partitions 1-3
[root@alpha-ops lib]# partx -a /dev/sda
partx: /dev/sda: error adding partitions 1-4

3.创建PV

# 创建
[root@alpha-ops lib]# pvcreate /dev/sda4 
  Physical volume "/dev/sda4" successfully created.

# 查看
[root@alpha-ops lib]# pvscan 
  PV /dev/sda3   VG centos          lvm2 [<29.00 GiB / 4.00 MiB free]
  PV /dev/sda4                      lvm2 [70.00 GiB]
  Total: 2 [<99.00 GiB] / in use: 1 [<29.00 GiB] / in no VG: 1 [70.00 GiB]

4.添加VG

# 将/dev/sda4加入 VG centos
[root@alpha-ops audit]# vgextend centos /dev/sda4
  Volume group "centos" successfully extended
[root@alpha-ops audit]# pvs
  PV         VG     Fmt  Attr PSize   PFree  
  /dev/sda3  centos lvm2 a--  <29.00g   4.00m
  /dev/sda4  centos lvm2 a--  <70.00g <70.00g

# 查看
[root@alpha-ops lib]# vgdisplay

5.扩容LV

# 1.查看
[root@alpha-ops audit]# lvs
  LV   VG     Attr       LSize  Pool Origin Data%  Meta%  Move Log Cpy%Sync Convert
  data centos -wi-ao----  7.99g                                                    
  root centos -wi-ao---- 20.00g                                                    
  swap centos -wi-ao----  1.00g                                        

# 2.+70G
[root@alpha-ops audit]# lvresize -L +70G /dev/centos/data
  Size of logical volume centos/data changed from 7.99 GiB (2046 extents) to 77.99 GiB (19966 extents).
  Logical volume centos/data successfully resized.

# 3.xfs_growfs Centos7 or resize2fs Centos6
# Centos7
$ xfs_growfs /dev/centos_alpha-docker-os/root  
# Centos6
$ resize2fs /dev/centos_alpha-docker-os/root

6.移除lv

[root@alpha-docker-os /]# lvremove /dev/centos_alpha-docker-os/data
Do you really want to remove active logical volume centos_alpha-docker-os/data? [y/n]: y
  Logical volume "data" successfully removed

7.创建lv

# 1.创建100G 
[root@alpha-ops /]# lvcreate -L 100G -n data centos_alpha-docker-os
  Logical volume "data" created.

# 2.格式化lv并挂载
[root@alpha-docker-os /]# mkfs.xfs /dev/centos_alpha-docker-os/data
[root@alpha-docker-os /]# mount -t xfs /dev/centos_alpha-docker-os/data /data

# 3.设置挂载
[root@alpha-docker-os /]# blkid /dev/sda1
/dev/sda1: UUID="91a3fb61-dcc2-46f6-867f-c333e60577a9" TYPE="xfs"
[root@alpha-docker-os /]# vim /etc/fstab

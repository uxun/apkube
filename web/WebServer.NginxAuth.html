Web Authentication

1.password tools

yum install httpd-tools               #适用centos
sudo apt-get install apache2-utils    #适用ubuntu

2.生成用户和密码

# user 
$ htpasswd -c /etc/nginx/htpasswd.users sammy
$ htpasswd /etc/nginx/htpasswd.users another_user

# modify password
$ htpasswd -b /etc/nginx/htpasswd.users kibana "69WshXsGZcLWEDopj?mg"

# new user
$ htpasswd -bc /etc/nginx/htpasswd.users kibana "69WshXsGZcLWEDopj?mg"

# Del user
$ htpasswd -D /etc/nginx/htpasswd.users kibana

3.Nginx basic_auth

server {
    listen 80;

    server_name logstash.uxun.com;

    auth_basic "Restricted Access";
    auth_basic_user_file /etc/nginx/htpasswd.users;

    allow 2.2.2.2/32;
#    deny all;

    location / {
        proxy_pass http://192.168.1.240:5601;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

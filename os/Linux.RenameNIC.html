Centos 7 Rename NIC

1.Modify the NIC name

cd /etc/sysconfig/network-scripts
[root@alpha-swzn-222 network-scripts]# mv ifcfg-ens160 ifcfg-eth0
[root@alpha-swzn-222 network-scripts]# mv ifcfg-ens192 ifcfg-eth1

# 修改ifcfg-{eth0,eth1} 配置
ifcfg-eth0
NAME=ens160  --> eth0
DEVICE=ens160 --> eth0

ifcfg-eth1
NAME=ens192  --> eth1
DEVICE=ens192 --> eth1

2.Modify grub

$ cat /etc/sysconfig/grub
GRUB_TIMEOUT=5
GRUB_DISTRIBUTOR="$(sed 's, release .*$,,g' /etc/system-release)"
GRUB_DEFAULT=saved
GRUB_DISABLE_SUBMENU=true
GRUB_TERMINAL_OUTPUT="console"
GRUB_CMDLINE_LINUX="crashkernel=auto rd.lvm.lv=centos_alpha-docker-os/root rd.lvm.lv=centos_alpha-docker-os/swap rhgb quiet"
GRUB_DISABLE_RECOVERY="true"

# 修改位置GRUB_CMDLINE_LINUX rhgb >
# 添加内容 net.ifnames=0 biosdevname=0
e.g：
GRUB_CMDLINE_LINUX="crashkernel=auto rd.lvm.lv=centos_alpha-docker-os/root rd.lvm.lv=centos_alpha-docker-os/swap rhgb net.ifnames=0 biosdevname=0 quiet"

3.生成grub启动菜单

[root@alpha-swzn-222 network-scripts]# grub2-mkconfig -o /boot/grub2/grub.cfg
